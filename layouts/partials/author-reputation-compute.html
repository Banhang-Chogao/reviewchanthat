= false -}} {{- end -}} {{- end -}} {{- if $allMet -}} {{- $unlocked = true -}} {{- end -}} {{- else if eq $rule.type "founding" -}} {{- if $isFounding -}} {{- $unlocked = true -}} {{- end -}} {{- end -}} {{- $badgeStates = $badgeStates | append (dict "badge" .
